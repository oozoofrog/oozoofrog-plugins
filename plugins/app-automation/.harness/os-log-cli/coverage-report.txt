## 커버리지 리포트 생성 명령 (실행 시)
swift test --package-path /PATH/os-log-cli --enable-code-coverage
xcrun llvm-cov report \
  $(find .build/debug -name 'OSLogCLIPackageTests' -not -name '*.o' | head -1) \
  -instr-profile $(find .build/debug/codecov -name 'default.profdata' | head -1) \
  --ignore-filename-regex '.build|Tests'
